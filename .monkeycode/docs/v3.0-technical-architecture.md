### 1. Notebook 响应式引擎

#### 1.1 依赖图 (DAG) 实现

```go
// backend/internal/service/notebook/dag.go

package notebook

import (
    "errors"
    "sync"
)

// Cell 代表 Notebook 中的一个单元格
type Cell struct {
    ID       string            `json:"id"`
    Type     string            `json:"type"` // sql, python, markdown, ai
    Code     string            `json:"code"`
    Output   interface{}       `json:"output,omitempty"`
    Deps     []string          `json:"deps"` // 依赖的单元格 ID
    Metadata map[string]string `json:"metadata,omitempty"`
}

// DAG 有向无环图
type DAG struct {
    mu      sync.RWMutex
    nodes   map[string]*Cell
    edges   map[string][]string // node -> children
    reverse map[string][]string // node -> parents
}

// NewDAG 创建新的依赖图
func NewDAG() *DAG {
    return &DAG{
        nodes:   make(map[string]*Cell),
        edges:   make(map[string][]string),
        reverse: make(map[string][]string),
    }
}

// AddCell 添加单元格
func (d *DAG) AddCell(cell *Cell) error {
    d.mu.Lock()
    defer d.mu.Unlock()
    
    // 检查循环依赖
    if err := d.wouldCreateCycle(cell.ID, cell.Deps); err != nil {
        return err
    }
    
    d.nodes[cell.ID] = cell
    
    // 更新边
    for _, dep := range cell.Deps {
        d.edges[dep] = append(d.edges[dep], cell.ID)
        d.reverse[cell.ID] = append(d.reverse[cell.ID], dep)
    }
    
    return nil
}

// GetExecutionOrder 获取执行顺序 (拓扑排序)
func (d *DAG) GetExecutionOrder() ([]string, error) {
    d.mu.RLock()
    defer d.mu.RUnlock()
    
    visited := make(map[string]bool)
    order := []string{}
    
    var visit func(id string) error
    visit = func(id string) error {
        if visited[id] {
            return nil
        }
        
        // 先访问所有依赖
        for _, parent := range d.reverse[id] {
            if err := visit(parent); err != nil {
                return err
            }
        }
        
        visited[id] = true
        order = append(order, id)
        return nil
    }
    
    // 遍历所有节点
    for id := range d.nodes {
        if !visited[id] {
            if err := visit(id); err != nil {
                return nil, err
            }
        }
    }
    
    return order, nil
}

// GetDependents 获取依赖某个单元格的所有单元格
func (d *DAG) GetDependents(cellID string) []string {
    d.mu.RLock()
    defer d.mu.RUnlock()
    
    dependents := []string{}
    queue := []string{cellID}
    visited := make(map[string]bool)
    
    for len(queue) > 0 {
        current := queue[0]
        queue = queue[1:]
        
        for _, child := range d.edges[current] {
            if !visited[child] {
                visited[child] = true
                dependents = append(dependents, child)
                queue = append(queue, child)
            }
        }
    }
    
    return dependents
}
```

#### 1.2 响应式执行引擎

```go
// backend/internal/service/notebook/engine.go

package notebook

import (
    "context"
    "fmt"
    "sync"
)

// Engine Notebook 执行引擎
type Engine struct {
    dag        *DAG
    executor   *Executor
    cache      *Cache
    mu         sync.RWMutex
    listeners  map[string][]func(string, interface{})
}

// NewEngine 创建新的执行引擎
func NewEngine() *Engine {
    return &Engine{
        dag:       NewDAG(),
        executor:  NewExecutor(),
        cache:     NewCache(),
        listeners: make(map[string][]func(string, interface{})),
    }
}

// ExecuteCell 执行单个单元格
func (e *Engine) ExecuteCell(ctx context.Context, cellID string) error {
    e.mu.RLock()
    cell, exists := e.dag.nodes[cellID]
    e.mu.RUnlock()
    
    if !exists {
        return fmt.Errorf("cell %s not found", cellID)
    }
    
    // 检查依赖是否都已执行
    for _, dep := range cell.Deps {
        if !e.isExecuted(dep) {
            return fmt.Errorf("dependency %s not executed", dep)
        }
    }
    
    // 执行单元格
    output, err := e.executor.Execute(ctx, cell)
    if err != nil {
        return err
    }
    
    // 更新单元格输出
    e.mu.Lock()
    cell.Output = output
    e.mu.Unlock()
    
    // 通知监听器
    e.notifyListeners(cellID, output)
    
    // 触发依赖单元格重新计算
    go e.recomputeDependents(ctx, cellID)
    
    return nil
}

// recomputeDependents 重新计算依赖的单元格
func (e *Engine) recomputeDependents(ctx context.Context, cellID string) {
    dependents := e.dag.GetDependents(cellID)
    
    for _, depID := range dependents {
        select {
        case <-ctx.Done():
            return
        default:
            if err := e.ExecuteCell(ctx, depID); err != nil {
                // 记录错误，继续执行其他单元格
                fmt.Printf("Error recomputing %s: %v\n", depID, err)
            }
        }
    }
}

// OnChange 注册单元格变更监听器
func (e *Engine) OnChange(cellID string, callback func(string, interface{})) {
    e.mu.Lock()
    defer e.mu.Unlock()
    
    e.listeners[cellID] = append(e.listeners[cellID], callback)
}

// notifyListeners 通知监听器
func (e *Engine) notifyListeners(cellID string, output interface{}) {
    e.mu.RLock()
    callbacks := e.listeners[cellID]
    e.mu.RUnlock()
    
    for _, callback := range callbacks {
        go callback(cellID, output)
    }
}
```

// backend/internal/service/semantic/layer.go

package semantic

import (
    "context"
    "fmt"
)

// SemanticLayer 语义层
type SemanticLayer struct {
    metrics      *MetricStore
    dimensions   *DimensionStore
    relationships *RelationshipStore
    rules        *BusinessRuleStore
}

// NewSemanticLayer 创建语义层
func NewSemanticLayer() *SemanticLayer {
    return &SemanticLayer{
        metrics:      NewMetricStore(),
        dimensions:   NewDimensionStore(),
        relationships: NewRelationshipStore(),
        rules:        NewBusinessRuleStore(),
    }
}

// ResolveQuery 解析自然语言查询
func (sl *SemanticLayer) ResolveQuery(ctx context.Context, question string) (*ResolvedQuery, error) {
    // Step 1: 使用 AI 解析意图
    intent, err := sl.parseIntent(ctx, question)
    if err != nil {
        return nil, err
    }
    
    // Step 2: 映射指标和维度
    resolved := &ResolvedQuery{
        Intent: intent,
    }
    
    // 映射指标
    for _, metricName := range intent.Metrics {
        metric, err := sl.metrics.GetByName(metricName)
        if err != nil {
            return nil, fmt.Errorf("metric %s not found", metricName)
        }
        resolved.Metrics = append(resolved.Metrics, metric)
    }
    
    // 映射维度
    for _, dimName := range intent.Dimensions {
        dim, err := sl.dimensions.GetByName(dimName)
        if err != nil {
            return nil, fmt.Errorf("dimension %s not found", dimName)
        }
        resolved.Dimensions = append(resolved.Dimensions, dim)
    }
    
    // Step 3: 应用业务规则
    for _, rule := range sl.rules.GetRelevantRules(resolved) {
        resolved = rule.Apply(resolved)
    }
    
    return resolved, nil
}

// GenerateSQL 生成 SQL
func (sl *SemanticLayer) GenerateSQL(resolved *ResolvedQuery) (string, error) {
    builder := NewSQLBuilder()
    
    // SELECT 子句
    for _, dim := range resolved.Dimensions {
        builder.Select(dim.ColumnName)
    }
    for _, metric := range resolved.Metrics {
        builder.Select(fmt.Sprintf("%s as %s", metric.Formula, metric.Name))
    }
    
    // FROM 子句
    tables := sl.collectTables(resolved)
    builder.From(tables[0])
    
    // JOIN 子句
    for i := 1; i < len(tables); i++ {
        relationship := sl.relationships.FindRelationship(tables[0], tables[i])
        if relationship != nil {
            builder.Join(relationship)
        }
    }
    
    // WHERE 子句
    if resolved.Filters != nil {
        builder.Where(resolved.Filters)
    }
    
    // GROUP BY 子句
    if len(resolved.Dimensions) > 0 {
        builder.GroupBy(resolved.Dimensions)
    }
    
    return builder.Build()
}

// backend/internal/service/agent/query.go

package agent

import (
    "context"
    "encoding/json"

// QueryAgent 查询代理
type QueryAgent struct {
    model LLM
}

// Intent 查询意图
type Intent struct {
    Type       string   `json:"type"` // query, compare, trend, breakdown
    Metrics    []string `json:"metrics"`
    Dimensions []string `json:"dimensions"`
    Filters    []Filter `json:"filters"`
    TimeRange  *TimeRange `json:"time_range,omitempty"`
}

// ParseIntent 解析查询意图
func (qa *QueryAgent) ParseIntent(ctx context.Context, question string) (*Intent, AgentLog, error) {
    prompt := fmt.Sprintf(`
解析用户的自然语言查询，提取意图。

查询问题：%s

请分析并返回以下信息:
1. 查询类型：query(简单查询), compare(对比), trend(趋势), breakdown(下钻)
2. 涉及的指标 (metrics)
3. 涉及的维度 (dimensions)
4. 过滤条件 (filters)
5. 时间范围 (time_range)

返回 JSON 格式。
`, question)
    
    response, err := qa.model.Generate(ctx, prompt)
    if err != nil {
        return nil, AgentLog{}, err
    }
    
    var intent Intent
    if err := json.Unmarshal([]byte(response), &intent); err != nil {
        return nil, AgentLog{}, err
    }
    
    log := AgentLog{
        Agent:  "QueryAgent",
        Action: "ParseIntent",
        Input:  map[string]interface{}{"question": question},
        Output: map[string]interface{}{"intent": intent},
    }
    
    return &intent, log, nil
}

  }
  
  return (
    <div className="notebook-editor">
      <Toolbar onAddCell={addCell} onRunAll={runAll} />
      
      <div className="cells-container">
        {cells.map((cell, index) => (
          <Cell
            key={cell.id}
            cell={cell}
            index={index}
            onUpdateCode={updateCellCode}
            onRun={() => runCell(cell.id)}
            onDelete={() => removeCell(cell.id)}
          />
        ))}
      </div>
      
      <AddCellButton onAdd={addCell} />
    </div>
  )
}

### 2. Cell 组件

```typescript
// frontend/app/notebook/cell.tsx

'use client'

import { useState } from 'react'
import { CodeEditor } from '@/components/shared/code-editor'
import { SQLEditor } from '@/components/shared/sql-editor'
import { AIAssistant } from '@/components/shared/ai-assistant'
import { Chart } from '@/components/shared/chart'

interface CellProps {
  cell: {
    id: string
    type: 'sql' | 'python' | 'markdown' | 'ai'
    code: string
    output: any
    deps: string[]
  }
  index: number
  onUpdateCode: (id: string, code: string) => void
  onRun: () => void
  onDelete: () => void
}

export function Cell({ cell, onUpdateCode, onRun, onDelete }: CellProps) {
  const [isRunning, setIsRunning] = useState(false)
  const [showAI, setShowAI] = useState(false)
  
  const handleRun = async () => {
    setIsRunning(true)
    try {
      await onRun()
    } finally {
      setIsRunning(false)
    }
  }
  
  const renderOutput = () => {
    switch (cell.type) {
      case 'sql':
      case 'python':
        if (Array.isArray(cell.output)) {
          return <DataTable data={cell.output} />
        }
        return <pre>{JSON.stringify(cell.output, null, 2)}</pre>
        
      case 'ai':
        return (
          <AIInsightCard
            insight={cell.output.insight}
            confidence={cell.output.confidence}
            reasoning={cell.output.reasoning}
          />
        )
        
      default:
        return null
    }
  }
  
  return (
    <div className="cell border rounded-lg p-4 my-2">
      <CellHeader
        type={cell.type}
        index={cell.index}
        deps={cell.deps}
        onRun={handleRun}
        onDelete={onDelete}
        onAI={() => setShowAI(!showAI)}
        isRunning={isRunning}
      />
      
      <div className="cell-input mt-2">
        {cell.type === 'sql' && (
          <SQLEditor
            value={cell.code}
            onChange={(code) => onUpdateCode(cell.id, code)}
          />
        )}
        
        {(cell.type === 'python' || cell.type === 'ai') && (
          <CodeEditor
            language="python"
            value={cell.code}
            onChange={(code) => onUpdateCode(cell.id, code)}
          />
        )}
        
        {cell.type === 'markdown' && (
          <MarkdownEditor
            value={cell.code}
            onChange={(code) => onUpdateCode(cell.id, code)}
          />
        )}
      </div>
      
      {showAI && cell.type !== 'markdown' && (
        <AIAssistant
          code={cell.code}
          context={cell.deps}
          onApply={(suggestion) => onUpdateCode(cell.id, suggestion)}
        />
      )}
      
      {cell.output && (
        <div className="cell-output mt-4 p-4 bg-gray-50 rounded">
          {renderOutput()}
        </div>
      )}
    </div>
  )
}
```
